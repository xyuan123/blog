    # 1.await一般用来修饰Promise对象,能直接返回Promise对象的结果

        function sleep(time) {
      return new Promise(reslove => {

    # await错误处理

    一个async修饰的函数,如果内部有多个await，任何一个await修饰的Promise状态变为reject，后续代码都不会继续执行

        async function demo() {
     await Promise.reject(new Error('fail'))
     await Promise.resolve('ok') // 不会执行
    }
    1

    # 解决方案

        async function demo() {
      await Promise.reject(new Error('fail')).catch(err => {
        console.log(err)
      })
      const res = await Promise.resolve('ok') // 会执行
      console.log(res) // ok
    }
    demo()

    # await注意点

    # 1. 处理await可能出现的错误

    用try catch块去包裹

        async function demo() {
      try {
        await Promise.reject(new Error('fail'))
      } catch (error) {
        console.log(error)  // Error: fail
      }
    }
    demo()
